export const getProduct = async (id: number): Promise<Product | null> => {
  if (typeof id !== "number" || !Number.isInteger(id)) {
    throw new Error(`Invalid product ID: ${JSON.stringify(id)}`);
  }
 
  const request = new sql.Request();

};
